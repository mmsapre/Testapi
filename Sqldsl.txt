grammar SqlDsl;

// Entry point for the query
query
    : SELECT selectList FROM tableName (joinClause)* whereClause? (LIMIT limitClause)? (OFFSET offsetClause)? EOF
    ;

// SELECT clause
selectList
    : column (',' column)*
    ;

// FROM clause
tableName
    : ID (AS? ID)?
    ;

// JOIN clause
joinClause
    : (INNER? JOIN | LEFT JOIN | RIGHT JOIN) tableName ON condition
    ;

// WHERE clause
whereClause
    : WHERE conditionGroup (logicalOperator conditionGroup)*
    ;

logicalOperator
    : AND
    | OR
    ;

// Condition groups (with parentheses support)
conditionGroup
    : condition
    | '(' conditionGroup (logicalOperator conditionGroup)* ')'
    ;

// UPDATED CONDITION RULE
condition
    : column OPERATOR value
    | column BETWEEN value AND value
    | column NOT BETWEEN value AND value
    ;

// Column definition
column
    : jsonbColumn
    | regularColumn
    ;

// Regular column
regularColumn
    : ID ('.' ID)?
    ;

// JSONB column
jsonbColumn
    : ID '.' ID jsonbAccess
    ;

// JSONB path
jsonbAccess
    : ('->>' | '->') STRING (jsonbAccess)?
    ;

// Values
value
    : STRING
    | NUMBER
    | BOOLEAN
    | jsonbValue
    ;

// JSON object
jsonbValue
    : '{' pair (',' pair)* '}'
    ;

pair
    : STRING ':' value
    ;

// Keywords
SELECT: 'SELECT';
FROM: 'FROM';
WHERE: 'WHERE';
JOIN: 'JOIN';
INNER: 'INNER';
LEFT: 'LEFT';
RIGHT: 'RIGHT';
ON: 'ON';
LIMIT: 'LIMIT';
OFFSET: 'OFFSET';
AS: 'AS';
AND: 'AND';
OR: 'OR';
BETWEEN: 'BETWEEN';
NOT: 'NOT';

// Operators
OPERATOR
    : '='
    | '>'
    | '<'
    | '>='
    | '<='
    | '<>'
    | '@>'
    | 'LIKE'
    ;

// Literals
STRING: '\'' .*? '\'';
NUMBER: [0-9]+;
BOOLEAN: 'TRUE' | 'FALSE';

// Identifiers
ID: [a-zA-Z_][a-zA-Z0-9_]*;

// Whitespace
WS: [ \t\r\n]+ -> skip;
