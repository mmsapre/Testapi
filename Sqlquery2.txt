grammar SqlQuery;

query
    : SELECT selectList FROM tableName (joinClause)* whereClause? (LIMIT limitClause)? (OFFSET offsetClause)? EOF
    ;

selectList
    : column (',' column)*
    ;

tableName
    : ID (AS? ID)?
    ;

joinClause
    : (INNER? JOIN | LEFT JOIN | RIGHT JOIN) tableName ON condition
    ;

whereClause
    : WHERE conditionGroup (logicalOperator conditionGroup)*
    ;

logicalOperator
    : AND
    | OR
    ;

// 🔥 LABELLED CONDITION GROUP
conditionGroup
    : condition                                   #singleCondition
    | '(' conditionGroup (logicalOperator conditionGroup)* ')'   #groupedCondition
    ;

// 🔥 LABELLED CONDITION
condition
    : column OPERATOR value                       #binaryCondition
    | column BETWEEN value AND value              #betweenCondition
    | column NOT BETWEEN value AND value          #notBetweenCondition
    ;

// 🔥 LABELLED COLUMN
column
    : jsonbColumn    #jsonbColExpr
    | regularColumn  #regularColExpr
    ;

regularColumn
    : ID ('.' ID)?
    ;

jsonbColumn
    : ID '.' ID jsonbAccess
    ;

jsonbAccess
    : ('->>' | '->') STRING (jsonbAccess)?
    ;

value
    : STRING
    | NUMBER
    | BOOLEAN
    | jsonbValue
    ;

jsonbValue
    : '{' pair (',' pair)* '}'
    ;

pair
    : STRING ':' value
    ;

// TOKENS
SELECT: 'SELECT';
FROM: 'FROM';
WHERE: 'WHERE';
JOIN: 'JOIN';
INNER: 'INNER';
LEFT: 'LEFT';
RIGHT: 'RIGHT';
ON: 'ON';
LIMIT: 'LIMIT';
OFFSET: 'OFFSET';
AS: 'AS';
AND: 'AND';
OR: 'OR';
BETWEEN: 'BETWEEN';
NOT: 'NOT';

OPERATOR
    : '=' | '>' | '<' | '>=' | '<=' | '<>' | '@>' | 'LIKE'
    ;

STRING: '\'' (~['\\] | '\\' .)* '\'';
NUMBER: [0-9]+;
BOOLEAN: 'TRUE' | 'FALSE';

ID: [a-zA-Z_][a-zA-Z0-9_]*;

WS: [ \t\r\n]+ -> skip;
